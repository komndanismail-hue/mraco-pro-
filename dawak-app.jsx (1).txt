function distKm(a, b) {
  const R = 6371, dLat = ((b.lat - a.lat) * Math.PI) / 180, dLng = ((b.lng - a.lng) * Math.PI) / 180;
  const s = Math.sin(dLat / 2) ** 2 + Math.cos((a.lat * Math.PI) / 180) * Math.cos((b.lat * Math.PI) / 180) * Math.sin(dLng / 2) ** 2;
  return R * 2 * Math.atan2(Math.sqrt(s), Math.sqrt(1 - s));
}

// ─── تحويل موقع من خرائط قوقل (Plus Code أو إحداثيات) بدون خدمات قوقل ───
const OLC_A = "23456789CFGHJMPQRVWX";
function olcEncode(lat, lng, len = 10) {
  let la = lat + 90, lo = lng + 180, code = "", res = 20;
  for (let i = 0; i < len / 2; i++) {
    code += OLC_A[Math.floor(la / res)] + OLC_A[Math.floor(lo / res)];
    la %= res; lo %= res; res /= 20;
  }
  return code;
}
function olcDecode(code) {
  code = code.toUpperCase().replace("+", "");
  let lat = -90, lng = -180, res = 20;
  for (let i = 0; i + 1 < Math.min(code.length, 10); i += 2) {
    lat += OLC_A.indexOf(code[i]) * res;
    lng += OLC_A.indexOf(code[i + 1]) * res;
    res /= 20;
  }
  res *= 20;
  if (code.length > 10) {
    const g = OLC_A.indexOf(code[10]);
    lat += Math.floor(g / 4) * (res / 5); lng += (g % 4) * (res / 4);
    return { lat: lat + res / 10, lng: lng + res / 8 };
  }
  return { lat: lat + res / 2, lng: lng + res / 2 };
}
function parseLocation(text) {
  text = (text || "").trim();
  // إحداثيات مباشرة: 15.5533, 32.5384
  const m = text.match(/(-?\d{1,2}\.\d+)[,\s]+(-?\d{1,3}\.\d+)/);
  if (m) return { lat: parseFloat(m[1]), lng: parseFloat(m[2]) };
  // رمز Plus Code: كامل أو مختصر (مثل HG3Q+89 الخرطوم)
  const pm = text.match(/([23456789CFGHJMPQRVWX]{2,8}\+[23456789CFGHJMPQRVWX]{2,3})/i);
  if (pm) {
    let code = pm[1].toUpperCase();
    const before = code.split("+")[0];
    if (before.length < 8) code = olcEncode(15.58, 32.53, 10).slice(0, 8 - before.length) + code;
    return olcDecode(code);
  }
  return null;
}

              {customLoc ? (
                <div className="locchip">
                  <span>✓ {customLoc.label}</span>
                  <button onClick={() => setCustomLoc(null)}>تغيير</button>
                </div>
              ) : (
                <>
                  <button className="gps" onClick={useGPS} disabled={gpsBusy}>
                    {gpsBusy ? "⏳ جارٍ تحديد موقعك..." : "📍 حدد موقعي تلقائياً (GPS)"}
                  </button>
                  <div className="locrow">
                    <input placeholder="أو الصق موقعك من خرائط قوقل" value={locInput}
                      onChange={(e) => setLocInput(e.target.value)} style={{ marginBottom: 0 }} />
                    <button className="locgo" onClick={applyLocInput} disabled={!locInput.trim()}>تحديد</button>
                  </div>
                  <div className="hint" style={{ textAlign: "right", margin: "6px 0 10px" }}>
                    من خرائط قوقل: اضغط مطولاً على موقعك ← انسخ الإحداثيات أو رمز الموقع (مثل HG3Q+89) والصقه هنا
                  </div>
                  <label>أو اختر الحي يدوياً</label>
                  <select value={area} onChange={(e) => { setArea(e.target.value); setSent(false); }}>
                    <option value="">— اختر الحي —</option>
                    {AREAS.map((a) => <option key={a.name} value={a.name}>{a.name}</option>)}
                  </select>
                </>
              )}
              <label>ابحث بالاسم أو بالصورة 📷</label>
              <div className="drugrow">
                <input placeholder="اسم الدواء: Amoxicillin 500mg" value={drug} onChange={(e) => setDrug(e.target.value)} />
                <label className="camb">📷<input type="file" accept="image/*" onChange={handlePhoto} style={{ display: "none" }} /></label>
              </div>
              {photo && (
                <div className="photo-row">
                  <img className="thumb" src={photo} alt="" onClick={() => setViewImg(photo)} />
                  <div style={{ flex: 1 }}>
                    <button className="ai" onClick={recognizeDrug} disabled={aiBusy}>
                      {aiBusy ? "⏳ جارٍ قراءة الصورة..." : "✨ تعرّف على اسم الدواء من الصورة"}
                    </button>
                    <button className="rm" onClick={() => setPhoto(null)}>إزالة الصورة</button>
                  </div>
                </div>
              )}
              <label>ملاحظة (اختياري)</label>
              <input placeholder="مثال: شراب للأطفال" value={note} onChange={(e) => setNote(e.target.value)} />
              <label>عنوانك للتوصيل</label>
              <textarea rows={2} placeholder="الحي، الشارع، أقرب معلم" value={address} onChange={(e) => setAddress(e.target.value)} />
              <label className="chk">
                <input type="checkbox" checked={isRx} onChange={(e) => setIsRx(e.target.checked)} />
                دواء يُصرف بروشتة / تأمين (الروشتة المختومة مرفقة بالصورة)
              </label>
            </div>

            {userLoc && (
              <div className="card">
                <h2>أقرب الصيدليات إليك</h2>
                <div className="hint">اختر حتى 3 صيدليات • 🗺️ الموقع • 💬 واتساب</div>
                {shown.map((p) => (
                  <div key={p.id} className={`ph ${selected.includes(p.id) ? "sel" : ""}`} onClick={() => toggleSelect(p.id)}>
                    <div className="info">
                      <div className="nm">{p.name}</div>
                      {p.area && <div className="ar">{p.area}</div>}
                    </div>
                    <span className="dist">{p.dist < 1 ? `${Math.round(p.dist * 1000)} م` : `${p.dist.toFixed(1)} كم`}</span>
                    {p.map && <a className="ico mp" href={p.map} target="_blank" rel="noreferrer" onClick={(e) => e.stopPropagation()}>🗺️</a>}
                    {p.wa && <a className="ico wa" href={waLink(p)} target="_blank" rel="noreferrer" onClick={(e) => e.stopPropagation()}>💬</a>}
                  </div>
                ))}
                {!showAll && nearby.length > 8 && <button className="more" onClick={() => setShowAll(true)}>عرض كل الصيدليات ({nearby.length})</button>}
                <button className="btn" disabled={!canSend} onClick={sendRequest}>إرسال الطلب {selected.length > 0 ? `(${selected.length})` : ""}</button>
                {!canSend && <div className="hint">أكمل اسم الدواء والعنوان واختر صيدلية واحدة على الأقل</div>}
              </div>
            )}

            {requests.length > 0 && (
              <div className="card">
                <h2>طلباتي</h2>
                {requests.map((r) => {
                  const p = ALL.find((x) => x.id === r.pharmacyId);
                  const s = STATUS[r.status];
                  return (
                    <div key={r.id} className="card req" style={{ marginBottom: 9 }}>
                      <div className="top">
                        <div>
                          <div className="drug">{r.drug} {r.photo ? "📎" : ""}</div>
                          <div className="meta">{p?.name}{r.isRx ? " • روشتة" : ""}</div>
                        </div>
                        <span className="badge" style={{ color: s.color, background: s.bg }}>{s.label}</span>
                      </div>
                      <div className="time">أُرسل الساعة {r.time}</div>
                    </div>
                  );
                })}
              </div>
            )}
          </>
        )}

        {/* ══════════ الصيدلي ══════════ */}
        {mode === "pharm" && loaded && !pharmProfile && !joinMode && (
          <div className="card">
            <h2>تسجيل الصيدلية 💊</h2>
            <label>اختر صيدليتك من القائمة</label>
            <select value={claimId} onChange={(e) => setClaimId(e.target.value)}>
              <option value="">— ابحث عن صيدليتك —</option>
              {ALL.map((p) => <option key={p.id} value={p.id}>{p.name}{p.area ? ` — ${p.area}` : ""}</option>)}
            </select>
            <label>اسم المسؤول</label>
            <input value={regName} onChange={(e) => setRegName(e.target.value)} placeholder="اسم الصيدلي / المالك" />
            <label>رقم الهاتف (للتحقق)</label>
            <input value={regPhone} onChange={(e) => setRegPhone(e.target.value)} placeholder="09XXXXXXXX" inputMode="tel" />
            <button className="btn" disabled={!claimId || !regName.trim() || regPhone.trim().length < 9}
              onClick={() => { const pp = { pharmacyId: claimId, name: regName.trim(), phone: regPhone.trim() }; setPharmProfile(pp); save("dawak-pharm", pp); }}>
              هذه صيدليتي — تسجيل
            </button>
            <div className="hint">صيدليتك غير موجودة في القائمة؟ <button className="link-a" onClick={() => setJoinMode(true)}>قدّم طلب انضمام</button></div>
          </div>
        )}

        {mode === "pharm" && joinMode && !pharmProfile && (
          <div className="card">
            <h2>طلب انضمام صيدلية جديدة</h2>
            {joinSent && <div className="toast">✓ تم إرسال طلبك! ستتم مراجعته من الإدارة والتواصل معك.</div>}
            <label>اسم الصيدلية</label>
            <input value={joinName} onChange={(e) => setJoinName(e.target.value)} placeholder="بدون كلمة صيدلية" />
            <label>رقم الهاتف</label>
            <input value={joinPhone} onChange={(e) => setJoinPhone(e.target.value)} placeholder="09XXXXXXXX" inputMode="tel" />
            <label>الحي / المنطقة</label>
            <input value={joinArea} onChange={(e) => setJoinArea(e.target.value)} placeholder="مثال: بحري - الحلفايا" />
            <label>رابط الموقع من خرائط قوقل (اختياري)</label>
            <input value={joinLink} onChange={(e) => setJoinLink(e.target.value)} placeholder="https://maps.app.goo.gl/..." />
            <button className="btn" disabled={!joinName.trim() || joinPhone.trim().length < 9} onClick={submitJoin}>إرسال طلب الانضمام</button>
            <div className="hint"><button className="link-a" onClick={() => { setJoinMode(false); setJoinSent(false); }}>→ رجوع للتسجيل</button></div>
          </div>
        )}

        {mode === "pharm" && pharmProfile && (
          <>
            <div className="card">
              <h2>📊 لوحة {ALL.find((p) => p.id === myId)?.name}</h2>
              <div className="stats">
                <div className="stat"><div className="n">{stats.total}</div><div className="l">طلبات اليوم</div></div>
                <div className="stat"><div className="n">{stats.confirmed}</div><div className="l">مؤكدة</div></div>
                <div className="stat"><div className="n">{stats.delivered}</div><div className="l">مبيعات ✓</div></div>
                <div className="stat"><div className="n">{stats.total ? Math.round((stats.confirmed / stats.total) * 100) : 0}%</div><div className="l">نسبة التوفر</div></div>
              </div>
              <button className="btn blue" onClick={endShift}>🕐 إنهاء الشفت وحفظ تقرير اليوم</button>
              {shifts.filter((s) => s.pharmacyId === myId).length > 0 && (
                <div style={{ marginTop: 12 }}>
                  <label>تقارير الشفتات السابقة</label>
                  {shifts.filter((s) => s.pharmacyId === myId).slice(0, 7).map((s, i) => (
                    <div key={i} className="shift">
                      <span>{s.date} — {s.time}</span>
                      <span><b>{s.delivered}</b> مبيعات من <b>{s.total}</b> طلبات</span>
                    </div>
                  ))}
                </div>
              )}
              <div className="hint"><button className="link-a" onClick={() => { setPharmProfile(null); save("dawak-pharm", null); }}>تسجيل خروج</button></div>
            </div>

            {incoming.length === 0 ? (
              <div className="card empty"><div className="big">📭</div>لا توجد طلبات واردة بعد.</div>
            ) : (
              incoming.map((r) => {
                const s = STATUS[r.status];
                return (
                  <div key={r.id} className="card req">
                    <div className="top">
                      <div>
                        <div className="drug">💊 {r.drug}</div>
                        {r.note && <div className="meta">ملاحظة: {r.note}</div>}
                      </div>
                      <span className="badge" style={{ color: s.color, background: s.bg }}>{s.label}</span>
                    </div>
                    <div className="meta">👤 {r.clientName || "عميل"} {r.clientPhone ? `— ${r.clientPhone}` : ""}</div>
                    <div className="meta">📍 العنوان: {r.address} ({r.area})</div>
                    {r.isRx && <span className="badge rx">🩺 روشتة / تأمين مرفقة</span>}
                    {r.photo && <img className="thumb big" src={r.photo} alt="مرفق" onClick={() => setViewImg(r.photo)} />}
                    <div className="time">وصل الساعة {r.time}</div>
                    {r.status === "pending" && (
                      <div className="acts">
                        <button className="ok" onClick={() => updateStatus(r.id, "available")}>✓ متوفر عندنا</button>
                        <button className="no" onClick={() => updateStatus(r.id, "unavailable")}>غير متوفر</button>
                      </div>
                    )}
                    {r.status === "available" && (
                      <div className="acts"><button className="go" onClick={() => updateStatus(r.id, "delivering")}>🛵 تم إرسال الطلب للتوصيل</button></div>
                    )}
                  </div>
                );
              })
            )}
          </>
        )}

        {/* ══════════ الإدارة ══════════ */}
        {mode === "admin" && !adminOk && (
          <div className="card">
            <h2>🔐 دخول الإدارة</h2>
            <label>الرقم السري</label>
            <input type="password" value={pin} onChange={(e) => setPin(e.target.value)} placeholder="••••" inputMode="numeric" />
            <button className="btn" onClick={() => { if (pin === savedPin) { setAdminOk(true); setPin(""); } else alert("رقم سري خاطئ"); }}>دخول</button>
          </div>
        )}

        {mode === "admin" && adminOk && (
          <>
            <div className="card">
              <h2>📈 نشاط التطبيق</h2>
              <div className="stats">
                <div className="stat"><div className="n">{allToday.length}</div><div className="l">طلبات اليوم</div></div>
                <div className="stat"><div className="n">{requests.length}</div><div className="l">إجمالي الطلبات</div></div>
                <div className="stat"><div className="n">{ALL.length}</div><div className="l">الصيدليات</div></div>
                <div className="stat"><div className="n">{apps.filter((a) => a.status === "pending").length}</div><div className="l">طلبات انضمام</div></div>
              </div>
              {topActive.length > 0 && (
                <div>
                  <label>أنشط الصيدليات (بعدد الطلبات)</label>
                  {topActive.map(([pid, n]) => (
                    <div key={pid} className="shift"><span>{ALL.find((p) => p.id === pid)?.name || pid}</span><b>{n} طلب</b></div>
                  ))}
                </div>
              )}
            </div>

            <div className="card">
              <h2>🏪 طلبات انضمام الصيدليات</h2>
              {apps.length === 0 ? <div className="empty">لا توجد طلبات انضمام بعد.</div> :
                apps.map((a) => (
                  <div key={a.id} className="card req" style={{ marginBottom: 9 }}>
                    <div className="top">
                      <div>
                        <div className="drug">صيدلية {a.name}</div>
                        <div className="meta">📞 {a.phone} — 📍 {a.area || "غير محدد"}</div>
                      </div>
                      <span className="badge" style={a.status === "approved" ? { color: "#0E7A5F", background: "#E3F3EC" } : a.status === "rejected" ? { color: "#A33A3A", background: "#FBE9E9" } : { color: "#B96A00", background: "#FDF1DE" }}>
                        {a.status === "approved" ? "مقبولة ✓" : a.status === "rejected" ? "مرفوضة" : "قيد المراجعة"}
                      </span>
                    </div>
                    {a.status === "pending" && (
                      <div className="acts">
                        <button className="ok" onClick={() => decideApp(a.id, true)}>✓ قبول وإضافة للتطبيق</button>
                        <button className="no" onClick={() => decideApp(a.id, false)}>رفض</button>
                      </div>
                    )}
                  </div>
                ))}
            </div>

            <div className="card">
              <h2>🛡️ الأمن السيبراني — حماية بياناتك</h2>
              <div className="sec-item"><span className="s">✅</span><span><b>إخفاء الأرقام:</b> أرقام الصيدليات لا تظهر أبداً في واجهة العميل — التواصل عبر أزرار فقط، فلا يمكن نسخ القائمة.</span></div>
              <div className="sec-item"><span className="s">✅</span><span><b>تسجيل إجباري:</b> لا بحث ولا تواصل بدون تسجيل بالاسم والرقم — أي محاولة جمع ممنهج تكون مُعرَّفة الهوية.</span></div>
              <div className="sec-item"><span className="s">✅</span><span><b>لوحة إدارة بكلمة سر:</b> البيانات الكاملة لا يراها غيرك.</span></div>
              <div className="sec-item"><span className="s">🔜</span><span><b>عند الربط بالخادم (Firebase):</b> حد أقصى للطلبات لكل مستخدم (Rate Limiting)، إخفاء قاعدة البيانات خلف قواعد أمنية، وكشف أنماط الجمع الآلي وحظرها تلقائياً.</span></div>
              <label style={{ marginTop: 12 }}>تغيير الرقم السري للإدارة</label>
              <input type="password" value={newPin} onChange={(e) => setNewPin(e.target.value)} placeholder="رقم سري جديد" />
              <button className="btn" disabled={newPin.length < 4} onClick={() => { setSavedPin(newPin); save("dawak-pin", newPin); setNewPin(""); alert("تم تغيير الرقم السري ✓"); }}>حفظ الرقم الجديد</button>
              <div className="hint"><button className="link-a" onClick={() => { setAdminOk(false); setMode("client"); }}>خروج من الإدارة</button></div>
            </div>
          </>
        )}
      </div>

      {viewImg && (
        <div className="modal" onClick={() => setViewImg(null)}>
          <img src={viewImg} alt="" />
        </div>
      )}
    </div>
  );
}
